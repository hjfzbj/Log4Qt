# Separate CMakeLists.txt for QString benchmark
# To use: copy this to CMakeLists.txt or add to existing one

find_package(Qt${QT_VERSION_MAJOR} COMPONENTS Test REQUIRED)

set(qstring_benchmark_SOURCES
    qstring_benchmark.cpp
    qstring_benchmark.h
)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
add_executable(qstring_benchmark ${qstring_benchmark_SOURCES})
target_link_libraries(qstring_benchmark log4qt Qt${QT_VERSION_MAJOR}::Test)

# Optional: Add as a test
add_test(NAME qstring_benchmark COMMAND $<TARGET_FILE:qstring_benchmark>)
